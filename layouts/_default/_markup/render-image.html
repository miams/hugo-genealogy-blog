{{- /* Markdown image render hook. - External URLs: plain with lazy loading. - Local images under /static/: with width/height from imageConfig. If a sibling .webp file exists at the same path, wrap in so modern browsers fetch the smaller WebP and others fall back to the original. - If the file is missing on disk, falls back to a plain so the build doesn't break on stale paths. */ -}} {{- $src := .Destination -}} {{- $alt := .Text -}} {{- $title := .Title -}} {{- if strings.HasPrefix $src "http" -}} {{- else -}} {{- $rel := strings.TrimPrefix "/" $src -}} {{- $diskPath := printf "static/%s" $rel -}} {{- $webpRel := replaceRE `\.(jpe?g|png)$` ".webp" $rel -}} {{- $webpDiskPath := printf "static/%s" $webpRel -}} {{- $hasWebp := and (ne $webpRel $rel) (os.FileExists $webpDiskPath) -}} {{- $webpUrl := printf "/%s" $webpRel -}} {{- if os.FileExists $diskPath -}} {{- $cfg := imageConfig $diskPath -}} {{- if $hasWebp -}} {{- else -}} {{- end -}} {{- else -}} {{- end -}} {{- end -}}
